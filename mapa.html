// Configuração do backend seguro const API_URL = 'https://meu-backend-fiscal.vercel.app/api/ocorrencias'; let mapa; let marcadores = []; // Função para inicializar o mapa function inicializarMapa() { console.log('Inicializando mapa...'); mapa = L.map('map').setView([-18.573, -39.732], 13); L.tileLayer('https://{s}.basemaps.cartocdn.com/light_all/{z}/{x}/{y}{r}.png', { attribution: '© OSM' }).addTo(mapa); console.log('Mapa inicializado!'); document.getElementById('stats').innerHTML = '🔄 Carregando ocorrências...'; } // Função para carregar ocorrências do backend seguro async function carregarOcorrencias() { const status = document.getElementById('statusFilter').value; console.log('Carregando ocorrências via backend seguro...'); try { const resposta = await fetch(API_URL); if (!resposta.ok) { throw new Error('Erro ao buscar dados'); } const data = await resposta.json(); console.log('Ocorrências encontradas:', data.length); document.getElementById('stats').innerHTML = `📌 ${data.length} ocorrências`; // Remover marcadores antigos marcadores.forEach(marker => mapa.removeLayer(marker)); marcadores = []; // Adicionar novos marcadores data.forEach(ocorrencia => { if (ocorrencia.latitude && ocorrencia.longitude) { let conteudo = `
📅 ${new Date(ocorrencia.data_criacao).toLocaleString('pt-BR')}

${ocorrencia.texto.substring(0, 150)}

`; if (ocorrencia.tags && ocorrencia.tags.length > 0) { conteudo += `
`; ocorrencia.tags.forEach(tag => { conteudo += `${tag}`; }); conteudo += `
`; } if (ocorrencia.foto) { conteudo += ``; } conteudo += `
🏷️ Status: ${ocorrencia.status}
`; const marcador = L.marker([ocorrencia.latitude, ocorrencia.longitude]) .bindPopup(conteudo) .addTo(mapa); marcadores.push(marcador); } }); if (marcadores.length > 0) { const bounds = L.latLngBounds(marcadores.map(m => m.getLatLng())); mapa.fitBounds(bounds, { padding: [50, 50] }); } } catch (erro) { console.error('Erro ao carregar:', erro); document.getElementById('stats').innerHTML = '❌ Erro: ' + erro.message; } } // Aguardar o DOM carregar document.addEventListener('DOMContentLoaded', () => { console.log('DOM carregado, iniciando...'); inicializarMapa(); setTimeout(() => { carregarOcorrencias(); }, 500); }); // Botão de atualizar document.getElementById('refreshBtn').addEventListener('click', carregarOcorrencias); document.getElementById('statusFilter').addEventListener('change', carregarOcorrencias); console.log('Script do mapa carregado com backend seguro!');
